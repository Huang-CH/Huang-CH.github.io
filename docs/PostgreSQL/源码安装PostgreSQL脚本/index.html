    源码安装PostgreSQL脚本 
    #! /bin/bash

# 作者：黄超
# 创建时间：2023年10月9日
# 脚本描述：此脚本用于 CentOS 7 静默安装 PostgreSQL

# 定义全局变量
PG_VERSION=14.10
PG_DIR=/opt/postgresql-${PG_VERSION}

ERROR_LOG_FILE=/var/log/CentOS_7_Install_PostgreSQL.log

# 定义函数，用于记录日志并退出脚本
exit_with_error() {
    echo "[ERROR] $1"
    echo "[ERROR] $1" >> ${ERROR_LOG_FILE}
    exit 1
}

# 创建 postgres 用户组与用户
groupadd -g 1000 postgres
useradd -g 1000 -u 1000 postgres

# 安装依赖
yum -y install gcc readline* zlib-devel python-devel openssl-devel libxml2-devel libxslt-devel systemd-devel

# 解压安装文件
tar -zxvf /opt/installed/postgresql-${PG_VERSION}.tar.gz -C /opt/

# 创建 PG 数据目录
mkdir -p /home/postgres/pgsql/{pg_home,pg_data,pg_archive,pg_wal,pg_backup,pg_log}

# 修改数据目录的属主和属组
chown -R postgres:postgres /home/postgres/pgsql

# 修改数据目录的权限
chmod -R 700 /home/postgres/pgsql

# 系统参数优化

# 调整系统资源限制

# 进入PostgreSQL安装目录
cd ${PG_DIR}

# 配置安装参数
./configure --prefix=/home/postgres/pgsql/pg_home --with-openssl --with-python --with-libxml --with-libxslt --enable-thread-safety --with-systemd || exit_with_error "编译配置出错！程序异常退出！"

# 编译并安装
make world && make install-world || exit_with_error "编译安装出错！程序异常退出！"

# 配置 postgres 用户的环境变量
su - postgres -c 'echo "export PGHOME=/home/postgres/pgsql/pg_home" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export PGDATA=/home/postgres/pgsql/pg_data" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export LANG=en_US.UTF8" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export LD_LIBRARY_PATH=/home/postgres/pgsql/pg_home/lib:${LD_LIBRARY_PATH}" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export DATE=\`date +"%Y%-m%-d %H:%M:%S"\`" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export PATH=/home/postgres/pgsql/pg_home/bin:$PATH" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export MANPATH=/home/postgres/pgsql/pg_home/share/man:${MANPATH}" >> /home/postgres/.bashrc'
su - postgres -c 'echo "export PGUSER=postgres" >> /home/postgres/.bashrc'

# 使环境变量生效
su - postgres -c 'source /home/postgres/.bashrc'

# 初始化数据库
su - postgres -c 'initdb -D /home/postgres/pgsql/pg_data -E UTF8 --locale=C -U postgres' || exit_with_error "postgres初始化失败！"

# 添加 PostgreSQL 的 systemd 服务配置文件
cat > /etc/systemd/system/postgresql.service <<EOF
[Unit]
Description=PostgreSQL Server
After=network.target

[Service]
Type=forking
ExecStart=/home/postgres/pgsql/pg_home/bin/pg_ctl -D /home/postgres/pgsql/pg_data start
User=postgres
Group=postgres

[Install]
WantedBy=multi-user.target
EOF

# 重载 systemd 服务配置文件
systemctl daemon-reload

# 启动数据库
# su - postgres -c  'pg_ctl -D ${PGDATA} -l logfile start' || exit_with_error "postgres启动失败！"
systemctl start postgresql || exit_with_error "PostgreSQL启动失败！"

# 添加开机启动项
systemctl enable postgresql || exit_with_error "PostgreSQL添加开机启动项失败！"

echo "PostgreSQL 安装完毕！请注意修改相关配置文件内容。"
