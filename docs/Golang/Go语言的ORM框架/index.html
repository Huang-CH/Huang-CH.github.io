    Go语言的ORM框架 #

    GORM简介 #

    GORM是一个使用Go语言编写的ORM框架。它文档齐全，对开发者友好，支持主流数据库。

    中文官方网站：https://gorm.io/zh_CN/

    官方中文文档：https://gorm.io/zh_CN/docs/

    GORM安装 # 

    // 安装 GORM 框架
go get -u gorm.io/gorm

// 安装 MySQL 数据库驱动
go get -u gorm.io/driver/mysql
// 安装 SQL Server 数据库驱动
go get -u gorm.io/driver/mssql
// 安装 PostgreSQL 数据库驱动
go get -u gorm.io/driver/postgres
// 安装 SQLite 数据库驱动
go get -u gorm.io/driver/sqlite
// 安装 Oracle 数据库驱动
go get github.com/CengSin/oracle

    GORM连接数据库 #

    GORM连接MySQL # 

    import (
  "gorm.io/gorm"
  "gorm.io/driver/mysql"
)

// 参考 https://github.com/go-sql-driver/mysql#dsn-data-source-name 获取详情
dsn := "username:password@tcp(127.0.0.1:3306)/dbname?charset=utf8mb4&parseTime=True&loc=Local"
db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
if err != nil {
    panic(err)
}
defer gormDB.close()

    想要正确的处理 time.Time ，您需要带上 parseTime 参数， ( 更多参数) 要支持完整的 UTF-8 编码，您需要将 charset=utf8 更改为 charset=utf8mb4 查看 此文章 获取详情

    MySQL 驱动程序提供了一些[高级配置]( GitHub - go-gorm/mysql: GORM mysql driver)可以在初始化过程中使用，例如：

    import (
  "gorm.io/driver/mysql"
  "gorm.io/gorm"
)

db, err := gorm.Open(mysql.New(mysql.Config{
  DSN: "username:password@tcp(127.0.0.1:3306)/dbname?charset=utf8&parseTime=True&loc=Local", // DSN data source name
  DefaultStringSize: 256, // string 类型字段的默认长度
  DisableDatetimePrecision: true, // 禁用 datetime 精度，MySQL 5.6 之前的数据库不支持
  DontSupportRenameIndex: true, // 重命名索引时采用删除并新建的方式，MySQL 5.7 之前的数据库和 MariaDB 不支持重命名索引
  DontSupportRenameColumn: true, // 用 `change` 重命名列，MySQL 8 之前的数据库和 MariaDB 不支持重命名列
  SkipInitializeWithVersion: false, // 根据当前 MySQL 版本自动配置
}), &gorm.Config{})

    GORM 允许通过 DriverName 选项自定义 MySQL 驱动，例如：

    import (
  "gorm.io/gorm"
  _ "example.com/my_mysql_driver"
)

db, err := gorm.Open(mysql.New(mysql.Config{
  DriverName: "my_mysql_driver",
  DSN: "username:password@tcp(localhost:9910)/dbname?charset=utf8&parseTime=True&loc=Local", // Data Source Name，参考 https://github.com/go-sql-driver/mysql#dsn-data-source-name
}), &gorm.Config{})

    GORM 允许通过一个现有的数据库连接来初始化 *gorm.DB

    import (
  "gorm.io/gorm"
  "database/sql"
)

sqlDB, err := sql.Open("mysql", "mydb_dsn")
gormDB, err := gorm.Open(mysql.New(mysql.Config{Conn: sqlDB,}), &gorm.Config{})

    GORM连接SQL Server # 

    import (  
    "gorm.io/gorm"
    "gorm.io/driver/sqlserver"
)

// 参考 https://github.com/denisenkom/go-mssqldb 获取详情
dsn := "sqlserver://username:password@localhost:9930?database=dbname"
db, err := gorm.Open(sqlserver.Open(dsn), &gorm.Config{})

    GORM连接PostgreSQL # 

    import (
  "gorm.io/gorm"
  "gorm.io/driver/postgres"
)

dsn := "host=localhost user=username password=password dbname=dbname port=9920 sslmode=disable TimeZone=Asia/Shanghai"
db, err := gorm.Open(postgres.Open(dsn), &gorm.Config{})

    GORM使用 pgx 作为 postgres 的 database/sql 驱动，默认情况下，它会启用 prepared statement 缓存，你可以这样禁用它：

    import (
  "gorm.io/gorm"
  "gorm.io/driver/postgres"
)

// https://github.com/go-gorm/postgres
db, err := gorm.Open(postgres.New(postgres.Config{
  DSN: "user=username password=password dbname=dbname port=9920 sslmode=disable TimeZone=Asia/Shanghai",
  PreferSimpleProtocol: true, // disables implicit prepared statement usage
}), &gorm.Config{})

    GORM连接SQLite # 

    import (
  "gorm.io/gorm"
  "gorm.io/driver/sqlite"
)

// github.com/mattn/go-sqlite3
db, err := gorm.Open(sqlite.Open("dbname.db"), &gorm.Config{})

    GORM操作数据库 # 

    package main

import (
	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

// 1.定义模型
type UserInfo struct {
	gorm.Model
	Username string
	Password string
	MobilPhone uint
	Status bool
	RoleID uint
}

func main() {
	// 2.连接 MySQL 数据库
	dsn := "root:Capjet.CN@tcp(10.0.0.100:3306)/testDB?charset=utf8mb4&parseTime=True&loc=Local"
	db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
	if err != nil {
		panic(err)
	}

	// 3.把模型与数据库中的表对应起来
	db.AutoMigrate(&UserInfo{})

	// 4.对数据库的增、删、改、查
}

    GORM操作数据库之增 # 

    package main

import (
	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

// 1.定义模型
type UserInfo struct {
	gorm.Model
	Username string
	Password string
	MobilPhone uint
	Status bool
	RoleID uint
}

func main() {
	// 2.连接 MySQL 数据库
	dsn := "root:Capjet.CN@tcp(10.0.0.100:3306)/testDB?charset=utf8mb4&parseTime=True&loc=Local"
	db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
	if err != nil {
		panic(err)
	}

	// 3.把模型与数据库中的表对应起来
	db.AutoMigrate(&UserInfo{})

	// 4.创建记录
	// 创建模型（结构体）实例
	user := UserInfo{Username: "HC", Password: "123456", MobilPhone: 13888888888, Status: true, RoleID: 1}
	// 在数据库中新增一条数据
	db.Create(&user)
}

    GORM操作数据库之查 #

    一般查询 # 

    package main

import (
	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

// 1.定义模型
type UserInfo struct {
	gorm.Model
	Username string
	Password string
	MobilPhone uint
	Status bool
	RoleID uint
}

func main() {
	// 2.连接 MySQL 数据库
	dsn := "root:Capjet.CN@tcp(10.0.0.100:3306)/testDB?charset=utf8mb4&parseTime=True&loc=Local"
	db, err := gorm.Open(mysql.Open(dsn), &gorm.Config{})
	if err != nil {
		panic(err)
	}

	// 3.把模型与数据库中的表对应起来
	db.AutoMigrate(&UserInfo{})

	// 4.一般查询
	var user UserInfo  // 定义变量 user 为 UserInfo 类型

	// 根据主键查询第一条记录
	db.First(&user)  // 将查询到的结果写入到变量 user 中，因此 user 需要取地址

	// 根据主键查询最后一条记录
	db.Last(&user)

	// 根据主键查询指定的某条记录(仅当主键为整型时可用)
	db.First(&user, 100)

	// 查询所有的记录
	var users []UserInfo
	db.Find(&users)
	// 查看查询所用的SQL语句
	db.Debug().Find(&users)
}
