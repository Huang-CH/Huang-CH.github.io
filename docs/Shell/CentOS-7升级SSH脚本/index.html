    CentOS 7升级SSH脚本 
    #! /bin/bash

# 作者：黄超
# 创建时间：2023年9月20日
# 脚本描述：此脚本用于升级CentOS 7 的OpenSSL和OpenSSH

# 定义全局变量
INSTALL_DIR=/opt/installed
LOG_FILE="openssh_upgrade.log"

# 定义函数，用于记录日志并退出脚本
exit_with_error() {
    echo "[ERROR] $1"
    echo "[ERROR] $1" >> ${INSTALL_DIR}/${LOG_FILE}
    exit 1
}

# 安装依赖组件
yum -y install gcc zlib-devel.x86_64 || exit_with_error "安装依赖组件失败"

echo "开始更新 OpenSSL"
echo "--------------------------------------------------"
sleep 3

# 定义变量
OPENSSL_VERSION=1.1.1w
OPENSSL_FILENAME=openssl-${OPENSSL_VERSION}.tar.gz
OPENSSL_DIR=openssl-${OPENSSL_VERSION}

# 备份 OpenSSL
mv /usr/bin/openssl /usr/bin/openssl.old || exit_with_error "备份/usr/bin/openssl失败"

# 进入安装目录
cd ${INSTALL_DIR}

# 解压 OpenSSL
tar -xvf ${INSTALL_DIR}/${OPENSSL_FILENAME} || exit_with_error "解压${OPENSSL_FILENAME}失败"

# 进入 OpenSSL 目录
cd ${INSTALL_DIR}/${OPENSSL_DIR} || exit_with_error "进入${OPENSSL_DIR}目录失败"

# OpenSSL 安装配置
./config || exit_with_error "OpenSSL配置失败"

# OpenSSL 编译
make || exit_with_error "OpenSSL编译失败"

# OpenSSL 安装
make install || exit_with_error "OpenSSL编译失败"

# 创建相关文件和文件夹的软连接
ln -s /usr/local/bin/openssl /usr/bin/openssl
ln -snf /usr/local/lib64/libssl.so.1.1 /usr/lib64/libssl.so
ln -snf /usr/local/lib64/libssl.so.1.1 /usr/lib64/libssl.so.1.1
ln -snf /usr/local/lib64/libcrypto.so.1.1 /usr/lib64/libcrypto.so
ln -snf /usr/local/lib64/libcrypto.so.1.1 /usr/lib64/libcrypto.so.1.1

# 刷新可共享的动态链接库
ldconfig || exit_with_error "ldconfig命令执行失败"

echo "开始更新 OpenSSH"
echo "--------------------------------------------------"
sleep 3

# 定义变量
OPENSSH_VERSION=9.4p1
OPENSSH_FILENAME=openssh-${OPENSSH_VERSION}.tar.gz
OPENSSH_DIR=openssh-${OPENSSH_VERSION}

# 备份 OpenSSH
cp /usr/bin/ssh /usr/bin/ssh.old || exit_with_error "/usr/bin/ssh文件备份失败"
cp /usr/sbin/sshd /usr/sbin/sshd.old || exit_with_error "/usr/sbin/sshd文件备份失败"
mv /etc/ssh /etc/ssh.old || exit_with_error "/etc/ssh目录备份失败" 

# 进入安装目录
cd ${INSTALL_DIR}

# 解压 OpenSSH
tar -xvf ${INSTALL_DIR}/${OPENSSH_FILENAME} || exit_with_error "解压${OPENSSH_FILENAME}失败"

 # 进入 OpenSSH 目录
cd ${INSTALL_DIR}/${OPENSSH_DIR} || exit_with_error "进入${INSTALL_DIR}/${OPENSSH_DIR}目录失败"

# OpenSSH 安装配置
./configure --prefix=/usr/ --sysconfdir=/etc/ssh --with-ssl-dir=/usr/local/ssl --with-zlib --with-md5-passwords || exit_with_error "OpenSSH安装配置失败"

# OpenSSH 编译
make || exit_with_error "OpenSSH编译失败"

# OpenSSH 安装
make install || exit_with_error "OpenSSH安装失败"

# 复制 OpenSSH 初始化文件
cp /opt/installed/openssh-9.4p1/contrib/redhat/sshd.init /etc/init.d/sshd 
cp /opt/installed/openssh-9.4p1/contrib/redhat/sshd.pam /etc/pam.d/sshd.pam
mv /usr/lib/systemd/system/sshd.service /usr/lib/systemd/system/sshd.service.old

# 修改 sshd 配置文件
sed -i 's/#PermitRootLogin prohibit-password/PermitRootLogin yes/g' /etc/ssh/sshd_config || exit_with_error "进入${OPENSSH_DIR}/${OPENSSH_FILENAME}目录失败" 

# 重载配置
systemctl daemon-reload

# 重启 sshd 服务
systemctl restart sshd

echo "SSH服务升级完成！"
