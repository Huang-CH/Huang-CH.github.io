    Debian系统常用配置 #

    apt命令 #

    命令 取代的命令 命令的功能
    apt install apt-get install 安装软件包
    apt remove apt-get remove 移除软件包
    apt purge apt-get purge 移除软件包及配置文件
    apt update apt-get update 刷新存储库索引
    apt upgrade apt-get upgrade 升级所有可升级的软件包
    apt autoremove apt-get autoremove 自动删除不需要的包
    apt full-upgrade apt-get dist-upgrade 在升级软件包时自动处理依赖关系
    apt search apt-cache search 搜索应用程序
    apt show apt-cache show 显示装细节
    apt list / 列出包含条件的包（已安装，可升级等）
    apt edit-sources / 编辑源列表

    解决控制台显示乱码问题 # 

    apt install locales
dpkg-reconfigure locales

    重新登录或重启系统后生效

    更换国内源 #

    备份源配置文件

    cp /etc/apt/sources.list /etc/apt/sources.list.bak

    修改源配置文件

    vi /etc/apt/sources.list
# 注释掉deb cdrom

    使配置生效

    sudo apt -y update

    卸载vim-tiny # 

    sudo apt remove -y vim-tiny

    安装常用软件 # 

    sudo apt install -y vim net-tools bash-completion lrzsz htop wget

    固定IP地址 #

    备份网卡配置文件

    cp /etc/network/interfaces  /etc/network/interfaces.bak

    编辑网卡配置文件

    vim /etc/network/interfaces

    auto ens33                   # 在系统启动的时候启动网络接口，无论网络接口有无连接(插入网线)，同时使ifup -a对其生效
# allow-hotplug ens33        # 只有当内核从该接口检测到热插拔事件后才启动该接口。
iface ens33 inet static      # static表示使用固定ip，dhcp表示使用动态ip
address 192.168.0.100        # 设置ip地址为：192.168.0.100
network 192.168.0.0
netmask 255.255.255.0        # 设置子网掩码为：255.255.255.0
broadcast 192.168.0.255
gateway 192.168.0.1          # 设置网关为：192.168.0.1

    备份DNS配置文件

    cp /etc/resolv.conf /etc/resolv.conf.bak

    编辑DNS配置文件

    vim /etc/resolv.conf

    search debian.local
nameserver 8.8.8.8          # 设置首选dns（一般在安装Debian的时候，系统已经自动设置好了，这一步可以不修改）
nameserver 8.8.4.4          # 设置备用dns

    停用网卡

    ifdown -a          # -a表示全部网卡

    启用网卡

    ifup -a            # -a表示全部网卡

    备份SSH服务配置文件

    编辑SSH服务配置文件

    vim /etc/ssh/sshd_config

    添加用户 #

    使用adduser添加用户，在创建时会提示输入密码，并在/home下创建用户目录

    adduser postgres          # 添加postgres用户

    使用useradd添加用户，没有密码，创建的用户是不能登录的，需要使用passwd设置密码，同时，不会在/home下创建用户目录。

    useradd postgres
passwd ******

    useradd is a low level utility for adding users. On Debian, administrators should usually use adduser(8) instead.

    删除用户 #

    使用deluser删除用户，会退出并删除所在的组，使用--remove-all-files可以删除用户拥有的所有文件

    deluser --remove-all-files postgres

    使用userdel删除用户，会退出并删除所在的组

    userdel postgres

    userdel is a low level utility for removing users. On Debian, administrators should usually use deluser(8) instead
